# WKV (Weighted Key-Value) Mechanism # The core innovation of RWKV wkv(t) = Σ_{i=1}^{t-1} e^{-(t-1-i)w + k_i} · v_i + e^{u+k_t} · v_t ───────────────────────────────────────────────────────── Σ_{i=1}^{t-1} e^{-(t-1-i)w + k_i} + e^{u+k_t}} # Where: # w = learned time decay (per-channel) # u = bonus for current token # k = key projection # v = value projection
def wkv_forward(self, k, v, state): """WKV computation with linear complexity.""" # Extract time decay and bonus w = self.time_decay # (d_heads,) u = self.time_bonus # (d_heads,) # LoRA-based per-token time decay td = self.lora_td(k) # (B, T, H) # Update state (RNN step) alpha = torch.exp(-w + td) state = state * alpha + torch.exp(k) * v # Compute output num = state + torch.exp(u + k) * v den = state.sum(-1) + torch.exp(u + k) return num / den
