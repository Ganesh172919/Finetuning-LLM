HNSW (Hierarchical Navigable Small World) is the algorithm behind most vector databases. It organizes vectors into a multi-layer graph where each layer has different connectivity.

🔑 Key Insight: Greedy Search with Hierarchical Navigation

Start at the top layer (entry point). Greedily move to the nearest neighbor. When you can't get closer, drop down one layer and continue. At Layer 0, find the actual k nearest neighbors. This is O(log N) instead of O(N) brute force!

💻 Implementation

import numpy as np import faiss class VectorDatabase: """Simple vector database using FAISS.""" def __init__(self, dim, index_type="hnsw"): self.dim = dim if index_type == "flat": self.index = faiss.IndexFlatIP(dim) elif index_type == "hnsw": self.index = faiss.IndexHNSWFlat(dim, 32) elif index_type == "ivf": quantizer = faiss.IndexFlatIP(dim) self.index = faiss.IndexIVFFlat(quantizer, dim, 100) def add(self, embeddings): """Add vectors to the index.""" self.index.add(embeddings.astype(np.float32)) def search(self, query, k=5): """Find k nearest neighbors.""" distances, indices = self.index.search( query.astype(np.float32), k ) return distances, indices def __len__(self): return self.index.ntotal # Usage db = VectorDatabase(dim=384, index_type="hnsw") embeddings = np.random.randn(10000, 384) db.add(embeddings) query = np.random.randn(1, 384) distances, indices = db.search(query, k=5) print(f"Found {len(indices[0])} nearest neighbors")
